local function loadSetting()
    if isfolder and isfile then
        if not isfolder(folderName) then makefolder(folderName) end
        if isfile(filePath) then
            return readfile(filePath) == "true"
        end
    end
    return true
end

local function saveSetting(state)
    if writefile then
        if not isfolder(folderName) then makefolder(folderName) end
        writefile(filePath, tostring(state))
    end
end

end

-- Исправленный прыжок: добавлена строгая проверка на текущее состояние
local function doJump()
    local char = Player.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum and hum.FloorMaterial ~= Enum.Material.Air then
            local currentState = hum:GetState()
            -- Проверяем, что персонаж не находится уже в процессе прыжка или падения
            if currentState ~= Enum.HumanoidStateType.Jumping and currentState ~= Enum.HumanoidStateType.Freefall then
                hum:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        end
    end
end

local jumpHolding = false

local function onJumpStart()
    if jumpHolding then return end
    jumpHolding = true
    task.spawn(function()
        while jumpHolding do
            doJump()
            task.wait(0.01)
        end
    end)
end

local function onJumpEnd()
    jumpHolding = false
end

    JumpButton.BackgroundTransparency = 0.5

local function setStandardControls(visible)
    pcall(function()
        local touchGui = Player.PlayerGui:FindFirstChild("TouchGui")
        if touchGui then
            touchGui.Enabled = visible
            -- Дополнительно принудительно возвращаем видимость элементам Roblox
            local frame = touchGui:FindFirstChild("TouchControlFrame")
            if frame then
                local jump = frame:FindFirstChild("JumpButton")
                if jump then jump.Visible = visible end
                local stick = frame:FindFirstChild("DynamicThumbstick")
                if stick then stick.Visible = visible end
            end
        end
    end)
end

local function toggleSystem(state)
    if not state then showNotification("Wait...") end
    isEnabled = state
    ScreenGui.Enabled = state
    saveSetting(state)
    -- При отключении возвращаем стандартное управление и делаем кнопки видимыми
    if not state then
        setStandardControls(true)
    end
end

Player.Chatted:Connect(function(msg)
    local m = msg:lower()
    if m == "on" then toggleSystem(true)
    elseif m == "off" then toggleSystem(false) end
end)

local dragging = false
local inputObject = nil
local moveVector = Vector3.new(0, 0, 0)
local startPos = Vector2.new(0, 0)
local movedOnce = false

CaptureArea.InputBegan:Connect(function(input)
    if not isEnabled then return end
    if dragging then return end
    if input.UserInputType ~= Enum.UserInputType.Touch
        and input.UserInputType ~= Enum.UserInputType.MouseButton1 then return end

    dragging = true
    movedOnce = false
    inputObject = input
    startPos = Vector2.new(input.Position.X, input.Position.Y)
    Base.Position = UDim2.new(0, startPos.X - 20, 0, startPos.Y - 20)
    Stick.Position = UDim2.new(0, startPos.X - 22.5, 0, startPos.Y - 22.5)
    Base.Visible = true
    Stick.Visible = true
    for _, d in pairs(dots) do
        d.Position = UDim2.new(0, startPos.X - (d.Size.X.Offset/2), 0, startPos.Y - (d.Size.Y.Offset/2))
        d.Visible = true
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if isEnabled and dragging and input == inputObject then
        local currentPos = Vector2.new(input.Position.X, input.Position.Y)
        local diff = currentPos - startPos

        if not movedOnce and diff.Magnitude > 2 then
            movedOnce = true
            CaptureArea.Active = false
        end

        if diff.Magnitude > 0 then
            local direction = diff.Unit
            Stick.Position = UDim2.new(0, currentPos.X - 22.5, 0, currentPos.Y - 22.5)
            for i, dot in pairs(dots) do
                local ratio = i / 7
                local dotPos = startPos + (diff * ratio)
                dot.Position = UDim2.new(0, dotPos.X - (dot.Size.X.Offset/2), 0, dotPos.Y - (dot.Size.Y.Offset/2))
            end
            moveVector = Vector3.new(direction.X, 0, direction.Y)
        end
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input == inputObject then
        dragging = false
        movedOnce = false
        inputObject = nil
        Base.Visible = false
        Stick.Visible = false
        for _, d in pairs(dots) do d.Visible = false end
        moveVector = Vector3.new(0, 0, 0)
        CaptureArea.Active = true
    end
end)

RunService.RenderStepped:Connect(function()
    -- Убиваем стандартное управление только когда включён
    if isEnabled then
        pcall(function()
            local touchGui = Player.PlayerGui:FindFirstChild("TouchGui")
            if touchGui then
                touchGui.Enabled = false
                local frame = touchGui:FindFirstChild("TouchControlFrame")
                if frame then
                    local jump = frame:FindFirstChild("JumpButton")
                    if jump then jump.Visible = false end
                    local stick = frame:FindFirstChild("DynamicThumbstick")

                end
            end
        end)
    end

    if isEnabled and Player.Character and Player.Character:FindFirstChild("Humanoid") then
        Player.Character.Humanoid:Move(moveVector, true)
    end
end)
